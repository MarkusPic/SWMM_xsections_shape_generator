def swmm_std_cross_sections(label, rel_with):
    height = np.linspace(0, 1, len(rel_with))
    cross_section = CrossSection(label, height=1)
    for x, y in zip(height, rel_with):
        if (y == 0) and (x in (0, 1)):
            continue
        cross_section.add(x, y / 2)

    return cross_section
